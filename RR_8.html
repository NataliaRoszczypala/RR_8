Suits¶

American TV Series¶

Suits is an American television drama series created by Aaron Korsh, which premiered on June 23, 2011, on the USA Network. It revolves around Mike Ross (Patrick J. Adams), who begins working as a law associate for Harvey Specter (Gabriel Macht), despite never attending law school. The show focuses on Harvey and Mike managing to close cases, while maintaining Mike's secret.

url = 'https://en.wikipedia.org/wiki/List_of_Suits_episodes#Series_overview' 
html = request.urlopen(url)
bs = BS(html.read(), 'html.parser')

# Navigating to the ratings table
ratings = bs.find('span', {'id':'Ratings'}).parent.find_next_sibling().find('table', {'class':'wikitable'}).tbody
rating_rows = ratings.find_all("tr")
all_seasons_ratings = []

#Iterating over rows and extracting ratings for each season
for row in rating_rows:
    td_elements = row.find_all("td")
    ratings = [float(td.text.strip()) for td in td_elements if td.text.strip() != '–']

    # Calculating the average rating for the season
    if ratings:
        average_rating = statistics.mean(ratings)
        all_seasons_ratings.append(average_rating)

# Printing the results
for season_number, average_rating in enumerate(all_seasons_ratings, start=1):
    print(f"Season {season_number}: Average rating = {average_rating:.2f}")

season_lengths = [12, 16, 16, 16, 16, 16, 16, 16, 10]

# Calculating average viewership for each season
start_index = 0
average_viewership = []

for length in season_lengths:
    ratings_list = merged_views[start_index:start_index + length]
    avg_viewership = np.mean(ratings_list)
    average_viewership.append(avg_viewership)
    start_index += length

# Season numbers
season_numbers = list(range(1, len(average_viewership) + 1))

# Creating the plot
plt.plot(season_numbers, average_viewership, marker='o', linestyle='-')
plt.title('Season-to-Season Changes in Viewership')
plt.xlabel('Season')
plt.ylabel('Average Viewership')
plt.grid(False)
